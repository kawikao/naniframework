  128.      * Walk top-level array elements and add lft and rgt rows for nested set.
  129.      * Nested child paths will be handled in _parseRows function

  131.      * @param array $arrayData

  135.     private function _buildTree(array $arrayData)
  136.     {
  137.         // create temporary root_node which will be removed later
  138.         $this->_rows[0] = array('id'=>0,
  139.                                 'lft' => $this->_options['startleft'], 
  140.                                 'rgt' => $this->_options['startleft'] + 1, 
  141.         );
  142.         try {
  143.             // loop through top-level nodes
  144.             foreach($arrayData[$this->_options['rowtag']] as $row) {
  145.                 $this->_parseRows($row);

  147.

  149.             array_shift($this->_rows);
  150.             array_walk($this->_rows, array(&$this,'_moveSetLeft'));
  151.  
  152.         } catch(Exception $e) {
  153.             // xml error
  154.             require_once 'Nani/Tree/Exception.php';

  159.     /**
  160.      * Get rows from current node and recursively calls itself for all child nodes
  161.      *
  162.      * @param array $row current row
  163.      * @param integer $parent parent index
  164.      * @return $this->_rows 
  165.      */
  166.     private function _parseRows(array $row, $parent=0)
  167.     {
  168.         // use parentidtag as result row index
  169.         if(!empty($this->_options['parentidtag'])) {
  170.             $parent = $row[$this->_options['parentidtag']];
  171.         }
  172.  
  173.         // initialize temporary row array
  174.         $tmpRow = array();
  175.  
  176.         // step through each tag of current row
  177.         foreach($row as $tag => $value) {
  178.             // skip over child paths
  179.             if($tag != $this->_options['rowtag']) {
  180.                 $tmpRow[$tag] = $value;
  181.             }
  182.         }
  183.          
  184.         // set temporary variable to parents rgt value
  185.         $rgtParent = $this->_rows[$parent]['rgt'];
  186.  
  187.         // expand tree to make room for new node by bumping rgt values by 2 where (rgt > rgtParent)
  188.         array_walk($this->_rows, array(&$this, '_moveSetRight'), $rgtParent);
  189.  
  190.         // set this nodes lft and rgt values
  191.         $lft = $rgtParent;
  192.         $rgt = $rgtParent + 1;
  193.  
  194.         // append to rows object using this nodes rowidtag as the index
  195.         $this->_rows[$tmpRow[$this->_options['rowidtag']]] = array_merge($tmpRow, array('lft'=>$lft, 'rgt'=>$rgt));
  196.  
  197.         // child paths were skipped above so, if this node contains children
  198.         if (isset($row[$this->_options['rowtag']])) {
  199.             // for each child path
  200.             foreach($row[$this->_options['rowtag']] as $child) {
  201.                 // recursively parse data
  202.                 $this->_parseRows($child, $row[$this->_options['rowidtag']]);
  203.             }
  204.         }
  205.     }
